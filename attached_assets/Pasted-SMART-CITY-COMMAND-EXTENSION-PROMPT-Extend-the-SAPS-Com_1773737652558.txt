💻 SAMPLE COMMAND HANDLER
async function controlTraffic(command: string) {
  const parsed = parseCommand(command);

  if (parsed.type === "HOLD_RED") {
    await fetch("/api/traffic/control", {
      method: "POST",
      body: JSON.stringify({
        intersections: parsed.targets,
        state: "RED",
        duration: 120 // seconds
      })
    });
  }

  if (parsed.type === "GREEN_CORRIDOR") {
    await fetch("/api/traffic/green-corridor", {
      method: "POST",
      body: JSON.stringify({
        route: parsed.route
      })
    });
  }
}
